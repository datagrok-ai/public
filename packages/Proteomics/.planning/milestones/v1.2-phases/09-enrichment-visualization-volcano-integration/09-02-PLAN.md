       ```typescript
       @grok.decorators.func({'top-menu': 'Proteomics | Visualize | Enrichment Charts...'})
       static async enrichmentCharts(): Promise<void> {
         const tv = grok.shell.tv;
         const df = tv?.dataFrame;
         if (!df) { grok.shell.warning('No table open'); return; }
         if (df.getTag('proteomics.enrichment') !== 'true') {
           grok.shell.warning('Run Enrichment Analysis first (Proteomics | Enrichment Analysis)');
           return;
         }
         // Find the protein DataFrame by proteomics.de_complete tag
         const proteinDf = grok.shell.tables.find((t: DG.DataFrame) => t.getTag('proteomics.de_complete') === 'true');
         if (!proteinDf) {
           grok.shell.warning('No protein table with DE results found. Enrichment charts will open without volcano linking.');
           // Still open visualization without cross-DF wiring
           openEnrichmentVisualization(df, df);
           return;
         }
         openEnrichmentVisualization(df, proteinDf);
       }
       ```

    This gives users two entry points per the locked decision:
    - Auto-open: enrichment analysis OK handler calls openEnrichmentVisualization
    - Re-open: Proteomics | Visualize | Enrichment Charts... menu item for existing enrichment tables
