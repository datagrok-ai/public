```
function groupMean(cols: DG.Column[], rowIdx: number): number {
  let sum = 0, n = 0;
  for (const c of cols) {
    if (c.isNone(rowIdx)) continue;
    sum += c.get(rowIdx) as number;
    n++;
  }
  return n > 0 ? sum / n : NaN;
}
```
