```
// Constants — Phase 17 reads these to extend the schema additively.
export const RUNS_CSV_PATH = 'System:AppData/Proteomics/spc/runs.csv';
export const RUNS_META_PATH = 'System:AppData/Proteomics/spc/runs-meta.json';
export const RUNS_CSV_SCHEMA_VERSION = '1';

// Canonical column order — locked at Phase 16. Phase 17 appends 'campaign_id' to this list.
export const RUNS_CSV_COLUMNS: readonly string[] = [
  'run_id',                  // UUID (crypto.randomUUID())
  'instrument_id',           // free-text, slugified before use as filename component
  'acquisition_datetime',    // ISO-8601 string (date-only acceptable; sort lex)
  'run_label',               // df.name at compute time
  'median_intensity',        // number
  'missing_pct',             // number (0..100)
  'control_corr',            // number (NaN allowed)
  'protein_count',           // number
  'status',                  // 'pass' | 'flagged' | 'out_of_control'
  'rules_tripped',           // JSON-stringified string[]
  'source_project_id',       // string | null
  'source_df_name',          // string
  'computed_at',             // ISO-8601 string
] as const;

// Types
export interface SpcRunRow {
  run_id: string;
  instrument_id: string;
  acquisition_datetime: string;
  run_label: string;
  median_intensity: number;
  missing_pct: number;
  control_corr: number;
  protein_count: number;
  status: 'pass' | 'flagged' | 'out_of_control';
  rules_tripped: string;     // JSON-stringified string[]
  source_project_id: string | null;
  source_df_name: string;
  computed_at: string;
}

export interface IterationTraceEntry {
  iter: number;
  dropped: string[];          // run_ids dropped this iteration
}

export interface LockedBaseline {
  instrument_id: string;
  locked_at: string;          // ISO-8601
  included_run_ids: string[];
  excluded_run_ids: string[];
  iteration_trace: IterationTraceEntry[];
  metrics: BaselineSnapshot;  // re-export from spc.ts
  rules_enabled: SpcRulesEnabled;  // re-export from spc.ts
}

// Slugify — borrows the charset rule from publishing/publish-state.ts:slugifyTarget
// but case is PRESERVED (instrument names like 'QExactive-01' must round-trip).
// Empty or path-traversal input → 'unnamed'.
export function slugifyInstrumentId(raw: string): string;

// CSV I/O — read-modify-write per Pattern 2 in RESEARCH.md. No append primitive.
// Upserts on composite key (instrument_id, acquisition_datetime). Second call with
// same key OVERWRITES the prior row. Sets run_id on insert; preserves run_id on overwrite.
export function appendRun(row: Omit<SpcRunRow, 'run_id'> & {run_id?: string}): Promise<SpcRunRow>;

// Reads runs.csv. If instrumentId is provided, filters to that instrument and sorts
// by acquisition_datetime ASCENDING (Pitfall 7 lever — never sort by computed_at).
// Returns empty array if runs.csv does not yet exist.
export function loadRuns(instrumentId?: string): Promise<SpcRunRow[]>;

// Baseline I/O
export function loadBaseline(instrumentId: string): Promise<LockedBaseline | null>;
export function saveBaseline(instrumentId: string, baseline: LockedBaseline): Promise<void>;

// Schema versioning — first-ever appendRun creates runs-meta.json; idempotent on subsequent calls.
export function readSchemaVersion(): Promise<string | null>;
export function writeSchemaVersion(version: string): Promise<void>;

// Iterative outlier removal (Pitfall 6 lever).
// values: input run series (one metric, one instrument) with run_id alignment.
// Returns: retained run_ids + iteration trace.
// CAP: maxIter=2 per CONTEXT.md D-04. Stops early if no outliers dropped in an iteration.
export function iterativeOutlierRemoval(
  series: Array<{run_id: string; value: number}>,
  maxIter?: number,  // default 2
): {retained_run_ids: string[]; excluded_run_ids: string[]; iteration_trace: IterationTraceEntry[]};

// Compute mean/SD per metric over a retained-run subset (called AFTER iterativeOutlierRemoval).
// Sample-SD (n-1 denominator) for consistency with downstream Nelson rule 3σ bounds.
export function computeBaselineStats(retainedRows: SpcRunRow[]): BaselineSnapshot;
```
