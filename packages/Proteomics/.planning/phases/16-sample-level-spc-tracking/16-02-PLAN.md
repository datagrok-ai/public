// Types
export interface RunMeta {
  instrument_id: string;
  acquisition_datetime: string;     // ISO-8601
}
export interface SpcMetrics {
  median_intensity: number;          // NaN-tolerant median across all log2-intensity cells
  missing_pct: number;               // % of (protein × sample) cells that are null, 0..100
  control_corr: number;              // mean of pairwise Pearson over Group 1 columns; NaN if k<2
  protein_count: number;             // rowCount minus all-missing-rows
  sample_count: number;              // group1.columns.length + group2.columns.length
  computed_at: string;               // ISO-8601
}
export type SpcStatus = 'pass' | 'flagged' | 'out_of_control';
export interface SpcRulesEnabled {
  median_intensity: Record<'nelson_1' | 'nelson_2' | 'nelson_3' | 'nelson_4' | 'nelson_5' | 'nelson_6' | 'nelson_7' | 'nelson_8', boolean>;
  missing_pct:      Record<'nelson_1' | 'nelson_2' | 'nelson_3' | 'nelson_4' | 'nelson_5' | 'nelson_6' | 'nelson_7' | 'nelson_8', boolean>;
  control_corr:     Record<'nelson_1' | 'nelson_2' | 'nelson_3' | 'nelson_4' | 'nelson_5' | 'nelson_6' | 'nelson_7' | 'nelson_8', boolean>;
  protein_count:    Record<'nelson_1' | 'nelson_2' | 'nelson_3' | 'nelson_4' | 'nelson_5' | 'nelson_6' | 'nelson_7' | 'nelson_8', boolean>;
}
export interface SpcRuleResult {
  status: SpcStatus;
  rulesTripped: string[];            // ['nelson_1@median_intensity', 'nelson_5@control_corr', ...]
}
export interface MetricBaseline {
  mean: number;
  sd: number;
}
export interface BaselineSnapshot {
  median_intensity: MetricBaseline;
  missing_pct: MetricBaseline;
  control_corr: MetricBaseline;
  protein_count: MetricBaseline;
}

// Constants — Wheeler 1992 for n=2 MR-chart
export const WHEELER_D2 = 1.128;
export const WHEELER_D4 = 3.267;

// Functions
export function computeSpcMetrics(df: DG.DataFrame, groups: GroupAssignment, runMeta: RunMeta): SpcMetrics;

// Evaluates Nelson rules 1-8 on a SERIES of historical values for one metric.
// `value` is the new run's metric value. `series` is the prior runs' values for the same metric.
// Returns ONE-METRIC partial; the caller aggregates across the 4 metrics into SpcRuleResult.
export function evaluateNelsonRulesForMetric(
  metric: 'median_intensity' | 'missing_pct' | 'control_corr' | 'protein_count',
  newValue: number,
  priorSeries: number[],
  baseline: MetricBaseline,
  enabled: SpcRulesEnabled['median_intensity'],
): string[];                          // returns list of rule trips for this metric, e.g. ['nelson_1@median_intensity']

// Aggregates over all 4 metrics. status: rule-1 trip → 'out_of_control'; any other → 'flagged'; none → 'pass'.
export function evaluateNelsonRules(
  newMetrics: SpcMetrics,
  priorSeries: { median_intensity: number[]; missing_pct: number[]; control_corr: number[]; protein_count: number[] },
  baseline: BaselineSnapshot,
  enabled: SpcRulesEnabled,
): SpcRuleResult;

// Tag + belt-and-braces column writer. Idempotent on re-run via ensureFreshFloat-style pattern.
export function setSpcStatus(df: DG.DataFrame, metrics: SpcMetrics, ruleResult: SpcRuleResult): void;
export function getSpcStatus(df: DG.DataFrame): { metrics: SpcMetrics; status: SpcStatus; rulesTripped: string[] } | null;

// Run-meta helpers (mirror setGroups/getGroups in experiment-setup.ts)
export function setRunMeta(df: DG.DataFrame, meta: RunMeta): void;
export function getRunMeta(df: DG.DataFrame): RunMeta | null;

    `computeSpcMetrics(df, groups, runMeta)`:
    - Resolve `g1Cols = groups.group1.columns.map(n => df.col(n)).filter(c => c !== null)` and same for g2.
    - All intensity column union = `g1Cols.concat(g2Cols)`. NaN-tolerant: skip cells where `col.isNone(i)`.
    - `median_intensity`: collect every non-null value from every intensity column into a flat array, sort, return middle (or mean of two middle). Even count → average of indices `(n-1)/2` and `n/2`.
    - `missing_pct`: count cells where `col.isNone(i)` is true across all (row × col), divide by `(rowCount × intensityCols.length)`, multiply by 100.
    - `control_corr` (per D-03): if `g1Cols.length < 2`, return NaN. Otherwise compute pairwise Pearson on every unordered pair `(i, j)` where `i < j`. For each pair: filter rows where BOTH cols are non-null. If fewer than 3 joint observations, that pair's contribution is NaN (skipped). Pearson: `r = Σ((x_i - mean_x)(y_i - mean_y)) / sqrt(Σ(x_i - mean_x)² × Σ(y_i - mean_y)²)`. Return mean of NON-NaN pair correlations, or NaN if every pair excluded.
    - `protein_count`: count rows where AT LEAST ONE intensity cell is non-null.
    - `sample_count`: `groups.group1.columns.length + groups.group2.columns.length`.
    - `computed_at`: `new Date().toISOString()`.

    `evaluateNelsonRulesForMetric(metric, newValue, priorSeries, baseline, enabled)`:
    - Build full series `s = priorSeries.concat([newValue])`. Sigma distances: `z_i = (s_i - baseline.mean) / baseline.sd`. If `baseline.sd === 0` or NaN, return [] (cannot evaluate).
    - Evaluate each rule against the FULL series (Nelson definitions need history). Trip the rule IFF the rule's pattern is satisfied at the END of the series (i.e., the new value participates).
    - Rule 1: `|z_last| > 3` → trip.
    - Rule 5: among the last 3 points, ≥ 2 have `z > 2` on the same side (both > 2 or both < -2) → trip.
    - Rules 2, 3, 4, 6, 7, 8 implemented per Nelson 1984 definitions (locked above). Each rule's check is at most ~10 LOC.
    - Filter trips to `enabled[ruleId]`. Return list of `'<ruleId>@<metric>'` strings.

    `evaluateNelsonRules(newMetrics, priorSeries, baseline, enabled)`:
    - For each of 4 metrics, call `evaluateNelsonRulesForMetric` and concat. Skip metrics whose newValue is NaN (D-03 — "metric unavailable" skip, not flag).
    - status: if ANY trip is `'nelson_1@*'` → `'out_of_control'`. Else if any trip → `'flagged'`. Else → `'pass'`.
